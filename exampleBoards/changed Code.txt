
printCell(Cell):-
    length(Cell,Size),
    nth1(Size,Cell,Color),
    symbol(Color,S),
    write(S),
    write('-'),
    write(Size).

        |
        |
        |
        V

printCell([X|L]) :- 
    symbol(X,S),
    write(S),
    write('-'),
    printCell(1,L).

printCell(N,[X|L]) :-
    M is N+1,
    printCell(M,L).

printCell(F,[]):-
    write(F).































    
    
getValue(Board,Player,ReturnValue,Value,CurrColumn,CurrRow):-

    length(Board, Size),
    write(CurrColumn-CurrRow-Size), write('\n'),
    write('Row>\n'),
    CurrRow>=0,
    
    

    getCellColor(Board,CurrColumn,CurrRow,Color),
    Color \= visited,
    Color = Player,
    Aux is CurrRow,
    replaceValue(Board,NewBoard,CurrColumn,Aux,Player),
    write(NewBoard), write('\n'),
    NewValue is Value + 1,

    /*up movement*/
    NewRow is CurrRow - 1,
    getValue(NewBoard, Player, ReturnValue,NewValue, CurrColumn, NewRow),
    
    ReturnValue is NewValue.
